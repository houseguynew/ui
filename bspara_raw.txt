local hasWriteFile = pcall(function() return writefile end) and type(writefile) == "function"
local hasReadFile = pcall(function() return readfile end) and type(readfile) == "function"

local function sanitizeKey(s)
	s = tostring(s or "")
	s = s:gsub("%s+", "_")
	s = s:gsub("[^%w_%-]", "")
	return s
end

local function new(className, props)
	local inst = Instance.new(className)
	if props then
		for k, v in pairs(props) do
			pcall(function()
				inst[k] = v
			end)
		end
	end
	return inst
end

local function udim2ToTable(u)
	if typeof(u) ~= "UDim2" then
		return nil
	end
	return {
		XScale = u.X.Scale,
		XOffset = u.X.Offset,
		YScale = u.Y.Scale,
		YOffset = u.Y.Offset,
	}
end

local function tableToUDim2(t)
	if type(t) ~= "table" then
		return nil
	end
	return UDim2.new(t.XScale or 0, t.XOffset or 0, t.YScale or 0, t.YOffset or 0)
end

local function prepareForSave(tbl)
	local out = {}
	for k, v in pairs(tbl or {}) do
		if typeof(v) == "UDim2" then
			out[k] = { _type = "UDim2", v = udim2ToTable(v) }
		elseif typeof(v) == "Vector2" then
			out[k] = { _type = "Vector2", v = { X = v.X, Y = v.Y } }
		elseif type(v) == "table" then
			out[k] = prepareForSave(v)
		else
			out[k] = v
		end
	end
	return out
end

local function restoreFromSave(tbl)
	if type(tbl) ~= "table" then
		return tbl
	end
	local out = {}
	for k, v in pairs(tbl) do
		if type(v) == "table" and v._type == "UDim2" and v.v then
			out[k] = tableToUDim2(v.v)
		elseif type(v) == "table" and v._type == "Vector2" and v.v then
			out[k] = Vector2.new(v.v.X or 0, v.v.Y or 0)
		elseif type(v) == "table" then
			out[k] = restoreFromSave(v)
		else
			out[k] = v
		end
	end
	return out
end

local function savePath(filename)
	return ("EnhancedUILib_%s.json"):format(sanitizeKey(filename or "default"))
end

local function loadFromFile(filename)
	if not hasReadFile then
		return nil
	end
	local ok, raw = pcall(function()
		return readfile(savePath(filename))
	end)
	if not ok or type(raw) ~= "string" or raw == "" then
		return nil
	end
	local decodedOk, decoded = pcall(function()
		return HttpService:JSONDecode(raw)
	end)
	if not decodedOk then
		return nil
	end
	return restoreFromSave(decoded)
end

local function writeToFile(filename, data)
	if not hasWriteFile then
		return false
	end
	local ok, encoded = pcall(function()
		return HttpService:JSONEncode(prepareForSave(data or {}))
	end)
	if not ok then
		return false
	end
	pcall(function()
		writefile(savePath(filename), encoded)
	end)
	return true
end

local function setCorner(parent, radius)
	new("UICorner", {
		Parent = parent,
		CornerRadius = radius or UDim.new(0, 12),
	})
end

local function setStroke(parent, color, transparency, thickness)
	new("UIStroke", {
		Parent = parent,
		Color = color or Color3.fromRGB(24, 24, 24),
		Transparency = transparency or 0,
		Thickness = thickness or 1,
	})
end

local function pulseHover(button, baseColor, hoverColor)
	local activeTween
	local function go(color)
		if activeTween then
			activeTween:Cancel()
		end
		activeTween = TweenService:Create(button, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			BackgroundColor3 = color,
		})
		activeTween:Play()
	end
	button.MouseEnter:Connect(function()
		go(hoverColor)
	end)
	button.MouseLeave:Connect(function()
		go(baseColor)
	end)
end

local function viewportSize()
	local cam = Camera or workspace.CurrentCamera or workspace:FindFirstChildOfClass("Camera")
	return (cam and cam.ViewportSize) or Vector2.new(1280, 720)
end

local function makeDraggable(handle, target)
	local dragging = false
	local dragInput
	local dragStart
	local startPos

	local function getPos(input)
		if input and input.Position then
			local p = input.Position
			return Vector2.new(p.X, p.Y)
		end
		return UserInputService:GetMouseLocation()
	end

	handle.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = getPos(input)
			startPos = target.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)

	handle.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging and input == dragInput then
			local delta = getPos(input) - dragStart
			local vs = viewportSize()
			local size = target.AbsoluteSize
			local anchor = target.AnchorPoint
			local topLeftX = startPos.X.Offset - (size.X * anchor.X)
			local topLeftY = startPos.Y.Offset - (size.Y * anchor.Y)

			local newTopLeftX = math.clamp(topLeftX + delta.X, 8, math.max(8, vs.X - size.X - 8))
			local newTopLeftY = math.clamp(topLeftY + delta.Y, 8, math.max(8, vs.Y - size.Y - 8))

			target.Position = UDim2.new(
				0,
				newTopLeftX + size.X * anchor.X,
				0,
				newTopLeftY + size.Y * anchor.Y
			)
		end
	end)
end

local function makeResizable(handle, target, opts)
	opts = opts or {}
	local dragging = false
	local dragInput
	local dragStart
	local startSize
	local startPos

	local function getPos(input)
		if input and input.Position then
			local p = input.Position
			return Vector2.new(p.X, p.Y)
		end
		return UserInputService:GetMouseLocation()
	end

	handle.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = getPos(input)
			startSize = target.Size
			startPos = target.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
					target:SetAttribute("ManualSized", true)
					if opts.onResizeEnd then
						pcall(function()
							opts.onResizeEnd(target)
						end)
					end
				end
			end)
		end
	end)

	handle.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging and input == dragInput then
			local delta = getPos(input) - dragStart
			local minW = opts.MinWidth or DEFAULTS.MinWidth
			local maxW = opts.MaxWidth or DEFAULTS.MaxWidth
			local minH = opts.MinHeight or DEFAULTS.MinHeight
			local maxH = opts.MaxHeight or DEFAULTS.MaxHeight

			local newW = math.clamp(startSize.X.Offset + delta.X, minW, maxW)
			local newH = math.clamp(startSize.Y.Offset + delta.Y, minH, maxH)

			local anchor = target.AnchorPoint
			local topLeft = Vector2.new(
				startPos.X.Offset - target.AbsoluteSize.X * anchor.X,
				startPos.Y.Offset - target.AbsoluteSize.Y * anchor.Y
			)

			local vs = viewportSize()
			local clampedX = math.clamp(topLeft.X, 8, math.max(8, vs.X - newW - 8))
			local clampedY = math.clamp(topLeft.Y, 8, math.max(8, vs.Y - newH - 8))

			target.Size = UDim2.new(0, newW, 0, newH)
			target.Position = UDim2.new(0, clampedX + newW * anchor.X, 0, clampedY + newH * anchor.Y)
		end
	end)
end

local function makeResponsive(frame, defaults)
	defaults = defaults or DEFAULTS
	local function update()
		if frame:GetAttribute("ManualSized") then
			return
		end
		local vs = viewportSize()
		local w = math.clamp(vs.X * defaults.WidthPercent, defaults.MinWidth, defaults.MaxWidth)
		local h = math.clamp(vs.Y * defaults.HeightPercent, defaults.MinHeight, defaults.MaxHeight)
		frame.Size = UDim2.new(0, w, 0, h)
		frame.Position = UDim2.new(0.5, -w / 2, 0.5, -h / 2)
	end
	update()
	local cam = Camera or workspace.CurrentCamera
	if cam then
		cam:GetPropertyChangedSignal("ViewportSize"):Connect(update)
	end
end

local function fitOnScreen(frame)
	local vs = viewportSize()
	local ap = frame.AbsolutePosition
	local asz = frame.AbsoluteSize
	local topLeftX = math.clamp(ap.X, 8, math.max(8, vs.X - asz.X - 8))
	local topLeftY = math.clamp(ap.Y, 8, math.max(8, vs.Y - asz.Y - 8))
	local anchor = frame.AnchorPoint
	frame.Position = UDim2.new(0, topLeftX + asz.X * anchor.X, 0, topLeftY + asz.Y * anchor.Y)
end

		Window = {},
		Controls = {},

		for k, v in pairs(loaded) do
			persistentState[k] = v

		if not hasWriteFile or saveQueued then
			return
		end
		saveQueued = true

			saveQueued = false
			pcall(function()
				writeToFile(filename, persistentState)
			end)
		end)
	end

		Name = "MainWindow",

		AnchorPoint = Vector2.new(0.5, 0.5),
		BackgroundColor3 = theme.Background,
		BorderSizePixel = 0,
	})
	setCorner(main, opts.CornerRadius or DEFAULTS.CornerRadius)
	setStroke(main, theme.Stroke, 0.1, 1)

	local shadow = new("Frame", {
		Parent = main,
		Name = "Shadow",
		BackgroundTransparency = 1,
		Size = UDim2.new(1, 22, 1, 22),
		Position = UDim2.new(0, -11, 0, -11),
		ZIndex = 0,
	})
	new("UICorner", { Parent = shadow, CornerRadius = opts.CornerRadius or DEFAULTS.CornerRadius })

		Parent = main,
		Name = "TopGlow",
		BackgroundColor3 = theme.Accent,
		BorderSizePixel = 0,
		BackgroundTransparency = 0.9,
		Size = UDim2.new(1, 0, 0, 2),
	})
	new("UIGradient", {
		Parent = topGlow,
		Color = ColorSequence.new({
			ColorSequenceKeypoint.new(0, theme.Accent),

		}),
		Transparency = NumberSequence.new({
			NumberSequenceKeypoint.new(0, 0.05),
			NumberSequenceKeypoint.new(1, 1),
		}),
	})

		Parent = main,
		Name = "TitleBar",
		Size = UDim2.new(1, 0, 0, 48),
		BackgroundColor3 = theme.Panel,
		BorderSizePixel = 0,
	})
	setCorner(titlebar, UDim.new(0, 16))

	local titleMask = new("Frame", {
		Parent = titlebar,
		BackgroundTransparency = 1,
		Size = UDim2.new(1, 0, 1, 0),
		ClipsDescendants = true,
	})

	local titleLabel = new("TextLabel", {
		Parent = titleMask,
		BackgroundTransparency = 1,
		Position = UDim2.new(0, 18, 0, 0),
		Size = UDim2.new(1, -220, 1, 0),
		Text = opts.Title or "Window",
		Font = Enum.Font.GothamSemibold,
		TextSize = 18,
		TextColor3 = theme.Text,
		TextXAlignment = Enum.TextXAlignment.Left,
	})

	local subtitle = new("TextLabel", {
		Parent = titleMask,
		BackgroundTransparency = 1,
		Position = UDim2.new(0, 18, 0, 23),
		Size = UDim2.new(1, -220, 0, 16),
		Text = opts.Subtitle or "Clean, responsive, and easier to use.",
		Font = Enum.Font.Gotham,
		TextSize = 12,
		TextColor3 = theme.Muted,
		TextXAlignment = Enum.TextXAlignment.Left,
	})

	local actionCluster = new("Frame", {
		Parent = titleMask,
		BackgroundTransparency = 1,
		AnchorPoint = Vector2.new(1, 0.5),
		Position = UDim2.new(1, -14, 0.5, 0),
		Size = UDim2.new(0, 128, 0, 32),
	})
	local actionLayout = new("UIListLayout", {
		Parent = actionCluster,
		FillDirection = Enum.FillDirection.Horizontal,
		HorizontalAlignment = Enum.HorizontalAlignment.Right,
		VerticalAlignment = Enum.VerticalAlignment.Center,
		Padding = UDim.new(0, 8),
	})

	local function makeActionButton(text, bg)
		local btn = new("TextButton", {
			Parent = actionCluster,
			Size = UDim2.new(0, 32, 0, 32),
			BackgroundColor3 = bg,
			Text = text,
			Font = Enum.Font.GothamBold,
			TextSize = 16,
			TextColor3 = theme.Text,
			AutoButtonColor = false,
			BorderSizePixel = 0,
		})
		setCorner(btn, UDim.new(0, 10))
		setStroke(btn, theme.Stroke, 0.2, 1)
		pulseHover(btn, bg, Color3.fromRGB(
			math.clamp(bg.R * 255 + 10, 0, 255),
			math.clamp(bg.G * 255 + 10, 0, 255),
			math.clamp(bg.B * 255 + 10, 0, 255)
		))
		return btn
	end

	local closeBtn = makeActionButton("×", theme.Danger)

		Parent = main,
		Name = "Body",
		BackgroundTransparency = 1,
		Position = UDim2.new(0, 0, 0, 48),
		Size = UDim2.new(1, 0, 1, -48),

		Parent = body,

		Font = Enum.Font.GothamSemibold,
		TextSize = 12,
		TextColor3 = theme.Muted,
		TextXAlignment = Enum.TextXAlignment.Left,
	})

	local tabsStrip = new("ScrollingFrame", {
		Parent = rail,
		BackgroundTransparency = 1,
		BorderSizePixel = 0,
		Position = UDim2.new(0, 0, 0, 28),
		Size = UDim2.new(1, 0, 1, -28),
		ScrollBarThickness = 4,
		CanvasSize = UDim2.new(0, 0, 0, 0),
	})
	local tabsList = new("UIListLayout", {
		Parent = tabsStrip,
		FillDirection = Enum.FillDirection.Vertical,
		SortOrder = Enum.SortOrder.LayoutOrder,
		Padding = UDim.new(0, 8),
	})
	local tabsCanvasConn = tabsList:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
		tabsStrip.CanvasSize = UDim2.new(0, 0, 0, tabsList.AbsoluteContentSize.Y + 8)
	end)

	local content = new("Frame", {
		Parent = body,
		Name = "Content",
		BackgroundTransparency = 1,
		Position = UDim2.new(0, 228, 0, 12),
		Size = UDim2.new(1, -240, 1, -24),
	})

	local contentCards = {}
	local window = {
		_gui = sg,
		_main = main,
		_tabs = {},
		theme = theme,
	}

	local function applySavedWindowState()
		if persistentState.Window then
			local w = persistentState.Window
			if typeof(w.Size) == "UDim2" then
				main.Size = w.Size
			end
			if typeof(w.Position) == "UDim2" then
				main.Position = w.Position
			end
			if typeof(w.AnchorPoint) == "Vector2" then
				main.AnchorPoint = w.AnchorPoint
			end
			if w.ManualSized then
				main:SetAttribute("ManualSized", true)
			end
		end
	end

	local function persistWindowState()
		persistentState.Window.Size = main.Size
		persistentState.Window.Position = main.Position
		persistentState.Window.AnchorPoint = main.AnchorPoint
		persistentState.Window.ManualSized = main:GetAttribute("ManualSized") and true or nil
		queueSave()
	end

	local isMinimized = false
	local savedState = {}

	local minimizedChip = new("TextButton", {
		Parent = sg,
		Name = "MinimizedChip",
		AnchorPoint = Vector2.new(0.5, 0),
		Position = UDim2.new(0.5, 0, 0, 10),
		Size = UDim2.new(0, 54, 0, 30),
		BackgroundColor3 = theme.PanelAlt,
		Text = "⌂",
		Font = Enum.Font.GothamBold,
		TextSize = 16,
		TextColor3 = theme.Text,
		Visible = false,
		AutoButtonColor = false,
		BorderSizePixel = 0,
	})
	setCorner(minimizedChip, UDim.new(0, 10))
	setStroke(minimizedChip, theme.Stroke, 0.15, 1)
	pulseHover(minimizedChip, theme.PanelAlt, theme.CardHover)

	local function topChipTarget(sizePx)
		local vs = viewportSize()
		local w, h = sizePx.X, sizePx.Y
		local topLeftX = vs.X * 0.5 - w * 0.5
		local topLeftY = 10
		return UDim2.new(0, topLeftX + w * 0.5, 0, topLeftY + h * 0.5)
	end

	local function minimize()
		if isMinimized then
			return
		end
		isMinimized = true
		savedState.Size = main.Size
		savedState.Position = main.Position
		savedState.AnchorPoint = main.AnchorPoint

		persistWindowState()

		local targetSize = UDim2.new(0, 54, 0, 30)
		local targetPos = topChipTarget(Vector2.new(54, 30))
		local tw = TweenService:Create(main, TweenInfo.new(0.20, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			Size = targetSize,
			Position = targetPos,
			BackgroundTransparency = 0,
		})
		tw:Play()
		tw.Completed:Connect(function()
			main.Visible = false
			minimizedChip.Visible = true
		end)
	end

	local function restore()
		if not isMinimized then
			return
		end
		isMinimized = false
		minimizedChip.Visible = false

		main.Visible = true
		main.Size = UDim2.new(0, 54, 0, 30)
		main.Position = topChipTarget(Vector2.new(54, 30))

		local tw = TweenService:Create(main, TweenInfo.new(0.24, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			Size = savedState.Size or UDim2.new(0, 740, 0, 520),
			Position = savedState.Position or UDim2.new(0.5, 0, 0.5, 0),
		})
		tw:Play()
		tw.Completed:Connect(function()
			if savedState.AnchorPoint then
				main.AnchorPoint = savedState.AnchorPoint
			end
			persistWindowState()
		end)
	end

	minimizeBtn.MouseButton1Click:Connect(function()
		if isMinimized then
			restore()
		else
			minimize()
		end
	end)

	minimizedChip.MouseButton1Click:Connect(restore)

	closeBtn.MouseButton1Click:Connect(function()
		writeToFile(filename, persistentState)
		sg:Destroy()
	end)

		Parent = main,
		AnchorPoint = Vector2.new(1, 1),
		Position = UDim2.new(1, -6, 1, -6),
		Size = UDim2.new(0, 20, 0, 20),
		BackgroundColor3 = theme.PanelAlt,
		Text = "↘",
		Font = Enum.Font.GothamBold,
		TextSize = 12,
		TextColor3 = theme.Muted,
		AutoButtonColor = false,
		BorderSizePixel = 0,
		ZIndex = 10,
	})
	setCorner(resizeGrip, UDim.new(0, 8))
	setStroke(resizeGrip, theme.Stroke, 0.2, 1)
	pulseHover(resizeGrip, theme.PanelAlt, theme.CardHover)
	makeResizable(resizeGrip, main, {
		MinWidth = opts.MinWidth or DEFAULTS.MinWidth,
		MaxWidth = opts.MaxWidth or DEFAULTS.MaxWidth,
		MinHeight = opts.MinHeight or DEFAULTS.MinHeight,

	local function selectTab(targetTab)
		window._activeTab = targetTab
		for _, tab in ipairs(window._tabs) do
			local active = tab == targetTab
			if tab.UpdateVisual then
				tab.UpdateVisual(active)
			else
				tab.Button.BackgroundColor3 = active and theme.Accent or theme.Card
				tab.Button.TextColor3 = active and Color3.fromRGB(255, 255, 255) or theme.Text
				local stroke = tab.Button:FindFirstChildOfClass("UIStroke")
				if stroke then
					stroke.Color = active and theme.Accent or theme.Stroke
					stroke.Transparency = active and 0 or 0.18
				end
			end
			tab.Content.Visible = active
		end
	end

			local function updatePopup()
				if not open then
					return
				end
				local absPos = card.AbsolutePosition
				local absSize = card.AbsoluteSize
				local vs = viewportSize()
				local width = math.max(200, absSize.X)
				local targetHeight = math.clamp(listLayout.AbsoluteContentSize.Y + 16, 0, 240)
				local desiredY = absPos.Y + absSize.Y + 8
				if desiredY + targetHeight > vs.Y - 8 then
					desiredY = math.max(8, absPos.Y - targetHeight - 8)
				end
				local desiredX = math.clamp(absPos.X, 8, math.max(8, vs.X - width - 8))
				popup.Position = UDim2.new(0, desiredX, 0, desiredY)
				popup.Size = UDim2.new(0, width, 0, targetHeight)
			end

			local function openPopup()
				rebuild()
				open = true
				popup.Visible = true
				updatePopup()
				trackingConn = RunService.RenderStepped:Connect(updatePopup)
			end

			local function closePopup()
				open = false
				popup.Visible = false
				if trackingConn then
					trackingConn:Disconnect()
					trackingConn = nil
				end
			end

			button.MouseButton1Click:Connect(function()
				if open then
					closePopup()
				else
					openPopup()
				end
			end)

			local inputConn
			inputConn = UserInputService.InputBegan:Connect(function(input)
				if not open then
					return
				end
				if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
					local p
					if input.Position then
						p = Vector2.new(input.Position.X, input.Position.Y)
					else
						p = UserInputService:GetMouseLocation()
					end
					local pos = popup.AbsolutePosition
					local size = popup.AbsoluteSize
					if p.X < pos.X or p.X > pos.X + size.X or p.Y < pos.Y or p.Y > pos.Y + size.Y then
						closePopup()
					end
				end
			end)

			local function setIndex(i)
				selectedIndex = math.clamp(i or 1, 1, math.max(1, #options))
				button.Text = tostring(options[selectedIndex] or "Select")
				rebuild()
				persistentState.Controls[generatedId] = selectedIndex
				queueSave()
				if type(callback) == "function" then
					pcall(function()
						callback(selectedIndex, options[selectedIndex])
					end)
				end
			end

			rebuild()

			return {
				Get = function()
					return selectedIndex, options[selectedIndex]
				end,
				SetIndex = function(i)
					setIndex(i)
				end,
				SetValue = function(value)
					for i, option in ipairs(options) do
						if option == value then
							setIndex(i)
							return
						end
					end
				end,
				UpdateOptions = function(newOptions, newDefaultIndex)
					options = newOptions or {}
					if newDefaultIndex ~= nil then
						selectedIndex = math.clamp(newDefaultIndex, 1, math.max(1, #options))
					else
						selectedIndex = math.clamp(selectedIndex, 1, math.max(1, #options))
					end
					button.Text = tostring(options[selectedIndex] or "Select")
					rebuild()
				end,
				_frame = card,
			}
		end

		table.insert(self._tabs, tab)
		if #self._tabs == 1 then
			selectTab(tab)
		end

		return tab
	end

	function window:Destroy()
		if self._gui and self._gui.Parent then
			writeToFile(filename, persistentState)
			self._gui:Destroy()
		end
	end

	main.Parent = sg
	main.BackgroundTransparency = 0
	TweenService:Create(main, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
		BackgroundTransparency = 0,
	}):Play()

	return window
end

return Library
