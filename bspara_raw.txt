-- ---------- Persistence using writefile (local JSON) ----------
local HttpService = game:GetService("HttpService")
local hasWriteFile, hasReadFile = pcall(function() return writefile end), pcall(function() return readfile end)
hasWriteFile = hasWriteFile and (type(writefile) == "function")
hasReadFile = hasReadFile and (type(readfile) == "function")

local function sanitizeKey(s)
    if not s then return "" end
    return tostring(s):gsub("%s+", "_"):gsub("[^%w_%-]", "")
end

local function udim2ToTable(u)
    if typeof(u) ~= "UDim2" then return nil end
    return { XScale = u.X.Scale, XOffset = u.X.Offset, YScale = u.Y.Scale, YOffset = u.Y.Offset }
end
local function tableToUDim2(t)
    if type(t) ~= "table" then return nil end
    return UDim2.new(t.XScale or 0, t.XOffset or 0, t.YScale or 0, t.YOffset or 0)
end

local function prepareForSave(tbl)
    local out = {}
    for k,v in pairs(tbl or {}) do
        if typeof(v) == "UDim2" then
            out[k] = { _type = "UDim2", v = udim2ToTable(v) }
        elseif type(v) == "table" then
            out[k] = prepareForSave(v)
        else
            out[k] = v
        end
    end
    return out
end

local function restoreFromSave(tbl)
    if type(tbl) ~= "table" then return tbl end
    local out = {}
    for k,v in pairs(tbl) do
        if type(v) == "table" and v._type == "UDim2" and v.v then
            out[k] = tableToUDim2(v.v)
        elseif type(v) == "table" then
            out[k] = restoreFromSave(v)
        else
            out[k] = v
        end
    end
    return out
end

local function getSavePath(filename)
    return ("SimpleUILib_%s.json"):format(sanitizeKey(tostring(filename or "default")))
end

local function loadFromFile(filename)
    if not hasReadFile then return nil end
    local path = getSavePath(filename)
    if not pcall(function() return readfile(path) end) then return nil end
    local ok, raw = pcall(function() return readfile(path) end)
    if not ok or not raw then return nil end
    local suc, json = pcall(function() return HttpService:JSONDecode(raw) end)
    if not suc then return nil end
    return restoreFromSave(json)
end

local function saveToFile(filename, tbl)
    if not hasWriteFile then return false end
    local path = getSavePath(filename)
    local safe = prepareForSave(tbl or {})
    local ok, encoded = pcall(function() return HttpService:JSONEncode(safe) end)
    if not ok then return false end
    pcall(function() writefile(path, encoded) end)
    return true
end
-- ---------- end writefile persistence ----------

local function new(class, props)
    local obj = Instance.new(class)
    if props then
        for k,v in pairs(props) do
            pcall(function() obj[k] = v end)
        end
    end
    return obj
end

local function makeDraggable(handle, target, opts)
    opts = opts or {}
    local margin = opts.margin or 40
    local snapToEdges = opts.snapToEdges
    local smooth = opts.smooth

    local dragging = false
    local dragInput, dragStart, dragOffset

    local function toVector2(v)
        if typeof(v) == "Vector2" then return v elseif typeof(v) == "Vector3" then return Vector2.new(v.X, v.Y) else return nil end
    end

    local function getInputPos(input)
        if input and input.Position then
            local p = toVector2(input.Position)
            if p then return p end
        end
        return UserInputService:GetMouseLocation()
    end

    local function safeAbsPositionAndSize()
        local ok1, absPos = pcall(function() return target.AbsolutePosition end)
        local ok2, absSize = pcall(function() return target.AbsoluteSize end)
        if ok1 and ok2 and absPos and absSize then return absPos, absSize end
        local px = (target.Position and target.Position.X and target.Position.X.Offset) or 0
        local py = (target.Position and target.Position.Y and target.Position.Y.Offset) or 0
        local sx = (target.Size and target.Size.X and target.Size.X.Offset) or 0
        local sy = (target.Size and target.Size.Y and target.Size.Y.Offset) or 0
        return Vector2.new(px, py), Vector2.new(sx, sy)
    end

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = getInputPos(input)
            local absPos, absSize = safeAbsPositionAndSize()
            dragOffset = dragStart - absPos

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                    if snapToEdges then
                        local vs = Camera and Camera.ViewportSize or Vector2.new(800,600)
                        local absPosNow, absSizeNow = safeAbsPositionAndSize()
                        local centerX = absPosNow.X + absSizeNow.X/2
                        local targetX = (centerX >= vs.X/2) and (vs.X - absSizeNow.X - margin) or margin
                        local posX = targetX + absSizeNow.X * target.AnchorPoint.X
                        local posY = absPosNow.Y + absSizeNow.Y * target.AnchorPoint.Y
                        local snapPos = UDim2.new(0, posX, 0, posY)
                        TweenService:Create(target, TweenInfo.new(0.18, Enum.EasingStyle.Quad), {Position = snapPos}):Play()
                    end
                end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local vs = Camera and Camera.ViewportSize or Vector2.new(800,600)
            local ipos = getInputPos(input)
            local desiredTopLeft = ipos - dragOffset
            local absPosNow, absSizeNow = safeAbsPositionAndSize()

            local clampedX = math.clamp(desiredTopLeft.X, margin * -1, vs.X - absSizeNow.X - margin)
            local clampedY = math.clamp(desiredTopLeft.Y, margin * -1, vs.Y - absSizeNow.Y - margin)

            local posX = clampedX + absSizeNow.X * target.AnchorPoint.X
            local posY = clampedY + absSizeNow.Y * target.AnchorPoint.Y
            local goal = UDim2.new(0, posX, 0, posY)

            if smooth then
                TweenService:Create(target, TweenInfo.new(0.05, Enum.EasingStyle.Linear), {Position = goal}):Play()
            else
                target.Position = goal
            end
        end
    end)
end

local function makeResizable(handle, target, opts)
    opts = opts or {}
    local dragging = false
    local dragInput, dragStart, startTopLeft, startSize

    local function toVector2(v)
        if typeof(v) == "Vector2" then return v end
        if typeof(v) == "Vector3" then return Vector2.new(v.X, v.Y) end
        return nil
    end

    local function getInputPos(input)
        if input and input.Position then
            local p = toVector2(input.Position)
            if p then return p end
        end
        return UserInputService:GetMouseLocation()
    end

    local function safeAbsPositionAndSize()
        local ok1, absPos = pcall(function() return target.AbsolutePosition end)
        local ok2, absSize = pcall(function() return target.AbsoluteSize end)
        if ok1 and ok2 and absPos and absSize then return absPos, absSize end
        local px = (target.Position and target.Position.X and target.Position.X.Offset) or 0
        local py = (target.Position and target.Position.Y and target.Position.Y.Offset) or 0
        local sx = (target.Size and target.Size.X and target.Size.X.Offset) or 0
        local sy = (target.Size and target.Size.Y and target.Size.Y.Offset) or 0
        return Vector2.new(px, py), Vector2.new(sx, sy)
    end

    local function ensureInViewport()
        local vs = Camera and Camera.ViewportSize or Vector2.new(800,600)
        local absPos, absSize = safeAbsPositionAndSize()
        local topLeft = absPos
        local w, h = absSize.X, absSize.Y
        local margin = opts.margin or 8
        local clampedX = math.clamp(topLeft.X, margin, math.max(0, vs.X - w - margin))
        local clampedY = math.clamp(topLeft.Y, margin, math.max(0, vs.Y - h - margin))
        if clampedX ~= topLeft.X or clampedY ~= topLeft.Y then
            local anchor = target.AnchorPoint or Vector2.new(0,0)
            local posX = clampedX + w * anchor.X
            local posY = clampedY + h * anchor.Y
            target.Position = UDim2.new(0, posX, 0, posY)
        end
    end

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = getInputPos(input)
            local absPos, absSize = safeAbsPositionAndSize()
            startTopLeft = absPos
            startSize = absSize
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                    target:SetAttribute("ManualSized", true)
                    if opts.onResizeEnd then pcall(function() opts.onResizeEnd(target) end) end
                    ensureInViewport()
                end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local ipos = getInputPos(input)
            local delta = ipos - dragStart

            local minW = opts.MinWidth or DEFAULTS.MinWidth
            local maxW = opts.MaxWidth or DEFAULTS.MaxWidth
            local minH = opts.MinHeight or DEFAULTS.MinHeight
            local maxH = opts.MaxHeight or DEFAULTS.MaxHeight

            local newW = math.clamp((startSize and startSize.X or 0) + delta.X, minW, maxW)
            local newH = math.clamp((startSize and startSize.Y or 0) + delta.Y, minH, maxH)

            local topLeft = startTopLeft or Vector2.new(0,0)
            local vs = Camera and Camera.ViewportSize or Vector2.new(800,600)
            local margin = opts.margin or 8
            local clampedTopLeftX = math.clamp(topLeft.X, margin, math.max(0, vs.X - newW - margin))
            local clampedTopLeftY = math.clamp(topLeft.Y, margin, math.max(0, vs.Y - newH - margin))

            local anchor = target.AnchorPoint or Vector2.new(0,0)
            local posX = clampedTopLeftX + newW * anchor.X
            local posY = clampedTopLeftY + newH * anchor.Y

            target.Size = UDim2.new(0, newW, 0, newH)
            target.Position = UDim2.new(0, posX, 0, posY)
        end
    end)
end

local function makeResponsive(frame, defaults)
    defaults = defaults or DEFAULTS
    local function update()
        if frame:GetAttribute("ManualSized") then return end

        local vs = Camera and Camera.ViewportSize or Vector2.new(800,600)
        local w = math.clamp(vs.X * defaults.WidthPercent, defaults.MinWidth, defaults.MaxWidth)
        local h = math.clamp(vs.Y * defaults.HeightPercent, defaults.MinHeight, defaults.MaxHeight)
        frame.Size = UDim2.new(0, w, 0, h)
        -- [FIXED] Perfectly center it based on the 0.5, 0.5 anchor point.
        frame.Position = UDim2.new(0.5, 0, 0.5, 0)
    end
    if Camera then
        Camera:GetPropertyChangedSignal("ViewportSize"):Connect(update)
    end
    update()
end

function Library.CreateWindow(opts)
    opts = opts or {}
    local accentColor = opts.AccentColor or DEFAULTS.AccentColor
    local sg = makeScreenGui(opts.Name or "SimpleUILib")

    local main = new("Frame", {
        Name = "MainWindow",
        Parent = sg,
        BackgroundColor3 = opts.BackgroundColor or Color3.fromRGB(27,27,27),
        BorderSizePixel = 0,
        AnchorPoint = Vector2.new(0.5, 0.5),
    })
    
    -- [Improvement 1] Setup UIScale for pop animation
    local uiScale = new("UIScale", {Parent = main, Scale = 0.85})
    new("UICorner", {Parent = main, CornerRadius = opts.CornerRadius or DEFAULTS.CornerRadius})
    new("UIStroke", {Parent = main, Color = Color3.fromRGB(18,18,18), Thickness = 1})

    local filename = sanitizeKey((opts and opts.SaveName) or (("player_%s_%s"):format(tostring(LocalPlayer.UserId), tostring(opts and opts.Name or "Window"))))
    local persistentState = { Window = {}, Controls = {} }

    -- Title bar 
    local titlebar = new("Frame", {Name = "TitleBar", Parent = main, Size = UDim2.new(1,0,0,36), BackgroundTransparency = 0.6, BackgroundColor3 = Color3.fromRGB(20,20,20), BorderSizePixel = 0})
    new("UICorner", {Parent = titlebar, CornerRadius = UDim.new(0,8)})
    local titleLabel = new("TextLabel", {Parent = titlebar, Size = UDim2.new(1,-12,1,0), Position = UDim2.new(0,12,0,0), BackgroundTransparency = 1, Text = opts.Title or "Window", TextColor3 = Color3.fromRGB(250,250,250), Font = DEFAULTS.FontBold, TextSize = 14, TextXAlignment = Enum.TextXAlignment.Left})

    local body = new("Frame", {Parent = main, Size = UDim2.new(1,0,1,-36), Position = UDim2.new(0,0,0,36), BackgroundTransparency = 1})

    -- Tabs strip 
    local tabsStrip = new("ScrollingFrame", {Parent = body, Size = UDim2.new(0,180,1,0), BackgroundColor3 = Color3.fromRGB(32,32,32), BorderSizePixel = 0, ScrollBarThickness = 4, ScrollBarImageColor3 = Color3.fromRGB(100,100,100)})
    new("UICorner", {Parent = tabsStrip, CornerRadius = UDim.new(0,8)})
    new("UIPadding", {Parent = tabsStrip, PaddingTop = UDim.new(0,8), PaddingLeft = UDim.new(0,8), PaddingRight = UDim.new(0,8)})
    tabsStrip.AutomaticCanvasSize = Enum.AutomaticSize.Y

    local tabsLayout = new("UIListLayout", {Parent = tabsStrip, FillDirection = Enum.FillDirection.Vertical, SortOrder = Enum.SortOrder.LayoutOrder})
    tabsLayout.Padding = UDim.new(0,8)

    local content = new("Frame", {Parent = body, Size = UDim2.new(1,-180,1,0), Position = UDim2.new(0,180,0,0), BackgroundTransparency = 1, Name = "Content"})
    new("UICorner", {Parent = content, CornerRadius = UDim.new(0,8)})

    local resizeGrip = new("Frame", {
        Name = "ResizeGrip", Parent = main, Size = UDim2.new(0, 18, 0, 18),
        Position = UDim2.new(1, -18, 1, -18), AnchorPoint = Vector2.new(0, 0),
        BackgroundColor3 = Color3.fromRGB(40,40,40), BorderSizePixel = 0,
    })
    new("UICorner", {Parent = resizeGrip, CornerRadius = UDim.new(0, 6)})

    local arrow = new("TextLabel", {
        Parent = resizeGrip, Size = UDim2.new(1, 0, 1, 0), BackgroundTransparency = 1,
        Text = "↘", Font = DEFAULTS.FontBold, TextSize = 14, TextColor3 = Color3.fromRGB(200,200,200),
    })

    makeResizable(resizeGrip, main, {
        MinWidth = DEFAULTS.MinWidth, MaxWidth = DEFAULTS.MaxWidth,
        MinHeight = DEFAULTS.MinHeight, MaxHeight = DEFAULTS.MaxHeight,
        smooth = true,
        onResizeEnd = function(targetFrame)
            persistentState.Window.ManualSized = true
            persistentState.Window.Size = targetFrame.Size
            persistentState.Window.Position = targetFrame.Position
            queueSave()
        end
    })

    -- Minimize UI
    local minimizeBtn = new("TextButton", {
        Parent = titlebar, Name = "MinimizeButton", Size = UDim2.new(0, 28, 0, 28), Position = UDim2.new(1, -36, 0, 4),
        BackgroundColor3 = Color3.fromRGB(50,50,50), BorderSizePixel = 0,
        Text = "-", Font = DEFAULTS.FontBold, TextSize = 14, TextColor3 = Color3.fromRGB(230,230,230),
    })
    new("UICorner", {Parent = minimizeBtn, CornerRadius = UDim.new(0,6)})

    local minimizedSquare = new("TextButton", {
        Parent = sg, Name = "MinimizedSquare", Size = UDim2.new(0, 44, 0, 44),
        AnchorPoint = Vector2.new(0.5, 0), Position = UDim2.new(0.5, 0, 0, 8), 
        BackgroundColor3 = Color3.fromRGB(40,40,40), BorderSizePixel = 0, Visible = false,
        Text = "↓", Font = DEFAULTS.FontBold, TextSize = 18, TextColor3 = Color3.fromRGB(230,230,230),
    })
    new("UICorner", {Parent = minimizedSquare, CornerRadius = UDim.new(0,8)})
    new("UIStroke", {Parent = minimizedSquare, Color = Color3.fromRGB(18,18,18), Thickness = 1})

    makeDraggable(minimizedSquare, minimizedSquare, {smooth = true})

    local _isMinimized = false
    local _savedState = {}

    local function topMiddleTarget(sizePx)
        local vs = Camera and Camera.ViewportSize or Vector2.new(800,600)
        local w, h = sizePx.X, sizePx.Y
        local topLeftX = vs.X * 0.5 - w * 0.5
        local topLeftY = 8
        local posX = topLeftX + w * main.AnchorPoint.X
        local posY = topLeftY + h * main.AnchorPoint.Y
        return UDim2.new(0, posX, 0, posY)
    end
    
    local loaded = loadFromFile(filename)
    if type(loaded) == "table" then
        for k,v in pairs(loaded) do persistentState[k] = v end
    end

    if persistentState.Window then
        persistentState.Window.Minimized = nil
    end

    if persistentState.Window then
        local W = persistentState.Window
        if W.Size and typeof(W.Size) == "UDim2" then pcall(function() main.Size = W.Size end) end
        if W.Position and typeof(W.Position) == "UDim2" then pcall(function() main.Position = W.Position end) end
        if W.AnchorPoint and typeof(W.AnchorPoint) == "Vector2" then pcall(function() main.AnchorPoint = W.AnchorPoint end) end
        if W.ManualSized then main:SetAttribute("ManualSized", true) end
        persistentState.Window.Minimized = nil
    end

    local saveQueued = false
    local saveDelay = 1.5 
    local function flushSave()
        saveQueued = false
        pcall(function() saveToFile(filename, persistentState) end)
    end
    local function queueSave()
        if not hasWriteFile then return end
        if saveQueued then return end
        saveQueued = true
        delay(saveDelay, flushSave)
    end

    Players.PlayerRemoving:Connect(function(p)
        if p == LocalPlayer and hasWriteFile then pcall(function() saveToFile(filename, persistentState) end) end
    end)
    if pcall(function() return game.BindToClose end) then
        pcall(function() game:BindToClose(function() if hasWriteFile then pcall(function() saveToFile(filename, persistentState) end) end end) end)
    end

    local function minimizeWindow()
        if _isMinimized then return end
        _isMinimized = true
        _savedState.Size = main.Size
        _savedState.Position = main.Position
        _savedState.AnchorPoint = main.AnchorPoint
        _savedState.ManualSized = main:GetAttribute("ManualSized")
    
        persistentState.Window.Size = _savedState.Size
        persistentState.Window.Position = _savedState.Position
        persistentState.Window.AnchorPoint = _savedState.AnchorPoint
        persistentState.Window.ManualSized = _savedState.ManualSized
        persistentState.Window.Minimized = nil
        queueSave()
    
        minimizeBtn.Active = false
        minimizedSquare.Active = false
    
        local squareW, squareH = 44, 44
        local targetSize = UDim2.new(0, squareW, 0, squareH)
        local targetPos = topMiddleTarget(Vector2.new(squareW, squareH))
    
        local tw = TweenService:Create(main, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = targetSize, Position = targetPos})
        tw:Play()
        tw.Completed:Connect(function()
            main.Visible = false
            minimizedSquare.Visible = true
            minimizeBtn.Active = true
            minimizedSquare.Active = true
        end)
    end

    local function restoreWindow()
        if not _isMinimized then return end
        _isMinimized = false
        minimizedSquare.Visible = false

        local targetSize = _savedState.Size or UDim2.new(0, 400, 0, 300)
        local targetPos  = _savedState.Position or UDim2.new(0.5, 0, 0.5, 0)
        local squareW, squareH = 44, 44
        local startSize = UDim2.new(0, squareW, 0, squareH)
        local startPos = topMiddleTarget(Vector2.new(squareW, squareH))

        main.Size = startSize
        main.Position = startPos
        main.Visible = true

        minimizeBtn.Active = false
        minimizedSquare.Active = false

        local tw = TweenService:Create(main, TweenInfo.new(0.28, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = targetSize, Position = targetPos})
        tw:Play()
        tw.Completed:Connect(function()
            if _savedState.ManualSized then
                main:SetAttribute("ManualSized", _savedState.ManualSized)
            else
                main:SetAttribute("ManualSized", nil)
            end
            persistentState.Window.Size = targetSize
            persistentState.Window.Position = targetPos
            persistentState.Window.AnchorPoint = _savedState.AnchorPoint
            persistentState.Window.ManualSized = _savedState.ManualSized
            persistentState.Window.Minimized = nil
            queueSave()
            minimizeBtn.Active = true
        end)
    end

    minimizeBtn.MouseButton1Click:Connect(minimizeWindow)
    minimizedSquare.MouseButton1Click:Connect(restoreWindow)
    minimizeBtn.MouseButton2Click:Connect(function() if _isMinimized then restoreWindow() else minimizeWindow() end end)

    -- [Improvement 2] Smooth Hover Animations added to styleButton
    local function styleButton(btn)
        btn.AutoButtonColor = false -- Turn off native to let tween handle it smoothly
        btn.Font = DEFAULTS.Font
        btn.TextSize = 13
        btn.TextColor3 = Color3.fromRGB(240,240,240)
        btn.BorderSizePixel = 0
        new("UICorner", {Parent = btn, CornerRadius = UDim.new(0,6)})
        
        -- Animation handlers
        local origColor = btn.BackgroundColor3
        local hoverColor = Color3.new(math.clamp(origColor.R + 0.1, 0, 1), math.clamp(origColor.G + 0.1, 0, 1), math.clamp(origColor.B + 0.1, 0, 1))
        
        btn.MouseEnter:Connect(function()
            TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundColor3 = hoverColor}):Play()
        end)
        btn.MouseLeave:Connect(function()
            TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundColor3 = origColor}):Play()
        end)
    end

    local function styleLabel(lbl)
        lbl.Font = DEFAULTS.Font
        lbl.TextSize = 13
        lbl.TextColor3 = Color3.fromRGB(240,240,240)
        lbl.BorderSizePixel = 0
        new("UICorner", {Parent = lbl, CornerRadius = UDim.new(0,6)})
    end

    makeDraggable(titlebar, main, {smooth = true})
    makeResponsive(main, opts.Responsive)

    pcall(function()
        local absPos, absSize = (function()
            local ok1, a = pcall(function() return main.AbsolutePosition end)
            local ok2, s = pcall(function() return main.AbsoluteSize end)
            if ok1 and ok2 then return a, s end
            return Vector2.new(main.Position.X.Offset, main.Position.Y.Offset), Vector2.new(main.Size.X.Offset, main.Size.Y.Offset)
        end)()
        local vs = Camera and Camera.ViewportSize or Vector2.new(800,600)
        local margin = 8
        local topLeft = absPos
        local w, h = absSize.X, absSize.Y
        local clampedX = math.clamp(topLeft.X, margin, math.max(0, vs.X - w - margin))
        local clampedY = math.clamp(topLeft.Y, margin, math.max(0, vs.Y - h - margin))
        if clampedX ~= topLeft.X or clampedY ~= topLeft.Y then
            main.Position = UDim2.new(0, clampedX + w * (main.AnchorPoint and main.AnchorPoint.X or 0), 0, clampedY + h * (main.AnchorPoint and main.AnchorPoint.Y or 0))
        end
    end)

    local window = {}
    window._gui = sg
    window._main = main
    window._tabs = {}

    local function selectTab(targetTab)
        for _, t in ipairs(window._tabs) do
            if t == targetTab then
                TweenService:Create(t.Button, TweenInfo.new(0.2), {BackgroundColor3 = accentColor, TextColor3 = Color3.fromRGB(255,255,255)}):Play()

                        if type(callback) == "function" then pcall(function() callback(state) end) end
                    end
                end,
                _frame = container,
            }

        function tab:AddDropdown(labelText, options, defaultIndex, id, callback)
            labelText = labelText or "Dropdown"
            options = options or {}
            defaultIndex = math.clamp((defaultIndex or 1), 1, math.max(1, #options))
       
            local container = new("Frame", {Parent = tabContent, Size = UDim2.new(1, 0, 0, 36), BackgroundTransparency = 1})
        
            local label = new("TextLabel", {
                Parent = container, Size = UDim2.new(1, -140, 1, 0), Position = UDim2.new(0, 8, 0, 0),
                BackgroundTransparency = 1, Text = labelText, Font = DEFAULTS.Font, TextSize = 13,
                TextColor3 = Color3.fromRGB(230,230,230), TextXAlignment = Enum.TextXAlignment.Left,
            })
        
            local currentBtn = new("TextButton", {
                Parent = container, Size = UDim2.new(0, 128, 0, 28), Position = UDim2.new(1, -136, 0.5, -14),
                BackgroundColor3 = Color3.fromRGB(60,60,60), BorderSizePixel = 0, Text = options[defaultIndex] or "—",
                TextScaled = false, AutoButtonColor = false,
            })
            new("UIPadding", {Parent = currentBtn, PaddingLeft = UDim.new(0,8), PaddingRight = UDim.new(0,24)})
            new("UICorner", {Parent = currentBtn, CornerRadius = UDim.new(0,6)})
            new("UIStroke", {Parent = currentBtn, Color = Color3.fromRGB(18,18,18), Thickness = 1})
            currentBtn.Font = DEFAULTS.Font
            currentBtn.TextSize = 13
            currentBtn.TextColor3 = Color3.fromRGB(240,240,240)
            currentBtn.TextXAlignment = Enum.TextXAlignment.Left
        
            local chevron = new("TextLabel", {
                Parent = currentBtn, Size = UDim2.new(0, 18, 1, 0), Position = UDim2.new(1, 4, 0, 0),
                BackgroundTransparency = 1, Text = "↓", Font = DEFAULTS.FontBold, TextSize = 14, TextColor3 = Color3.fromRGB(200,200,200),
            })
            
            local origDropColor = currentBtn.BackgroundColor3
            local hoverDropColor = Color3.new(math.clamp(origDropColor.R + 0.1, 0, 1), math.clamp(origDropColor.G + 0.1, 0, 1), math.clamp(origDropColor.B + 0.1, 0, 1))
            currentBtn.MouseEnter:Connect(function() TweenService:Create(currentBtn, TweenInfo.new(0.15), {BackgroundColor3 = hoverDropColor}):Play() end)
            currentBtn.MouseLeave:Connect(function() TweenService:Create(currentBtn, TweenInfo.new(0.15), {BackgroundColor3 = origDropColor}):Play() end)
        
            local sg = window._gui 
            local optionsFrame = new("Frame", {
                Parent = sg, Size = UDim2.new(0, 200, 0, 0), Position = UDim2.new(0, 0, 0, 0),
                AnchorPoint = Vector2.new(0,0), BackgroundColor3 = Color3.fromRGB(40,40,40),
                BorderSizePixel = 0, Visible = false, ClipsDescendants = true, ZIndex = 50,
            })
            new("UICorner", {Parent = optionsFrame, CornerRadius = UDim.new(0,6)})
            new("UIStroke", {Parent = optionsFrame, Color = accentColor, Thickness = 1}) -- [Improvement 5] Accent color border on dropdowns
            
            local optionsPadding = new("UIPadding", {Parent = optionsFrame, PaddingTop = UDim.new(0,6), PaddingBottom = UDim.new(0,6), PaddingLeft = UDim.new(0,6), PaddingRight = UDim.new(0,6)})
            local optionsList = new("ScrollingFrame", {
                Parent = optionsFrame, Size = UDim2.new(1, - (optionsPadding.PaddingLeft.Offset + optionsPadding.PaddingRight.Offset), 1, - (optionsPadding.PaddingTop.Offset + optionsPadding.PaddingBottom.Offset)),
                Position = UDim2.new(0, optionsPadding.PaddingLeft.Offset, 0, optionsPadding.PaddingTop.Offset),
                BackgroundTransparency = 1, ScrollBarThickness = 4, CanvasSize = UDim2.new(0,0,0,0),
                VerticalScrollBarInset = Enum.ScrollBarInset.Always, ZIndex = 51, ScrollBarImageColor3 = Color3.fromRGB(100,100,100)
            })
            local listLayout = new("UIListLayout", {Parent = optionsList, FillDirection = Enum.FillDirection.Vertical, SortOrder = Enum.SortOrder.LayoutOrder})
            listLayout.Padding = UDim.new(0,6)
            
            listLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
                optionsList.CanvasSize = UDim2.new(0,0,0, listLayout.AbsoluteContentSize.Y + 6)
            end)
            
            local selectedIndex = defaultIndex
            local generatedId = sanitizeKey(id or ((opts.Name or "Window") .. "_" .. (tab.Name or "Tab") .. "_" .. labelText))
            if persistentState.Controls[generatedId] ~= nil then selectedIndex = persistentState.Controls[generatedId] or selectedIndex end
            currentBtn.Text = tostring(options[selectedIndex] or currentBtn.Text)
            local optionButtons = {}

            local function rebuildOptions()
                for _, b in ipairs(optionButtons) do if b and b.Parent then b:Destroy() end end
                optionButtons = {}
                for i, v in ipairs(options) do
                    local opt = new("TextButton", {
                        Parent = optionsList, Size = UDim2.new(1, 0, 0, 32),
                        BackgroundColor3 = (i == selectedIndex) and accentColor or Color3.fromRGB(55,55,55),
                        BorderSizePixel = 0, Text = tostring(v or ""), ZIndex = 52,
                    })
                    opt.Font = DEFAULTS.Font
                    opt.TextSize = 13
                    opt.TextColor3 = Color3.fromRGB(240,240,240)
                    new("UICorner", {Parent = opt, CornerRadius = UDim.new(0,6)})
                    
                    if i ~= selectedIndex then
                        local optHover = Color3.new(math.clamp(opt.BackgroundColor3.R + 0.1, 0, 1), math.clamp(opt.BackgroundColor3.G + 0.1, 0, 1), math.clamp(opt.BackgroundColor3.B + 0.1, 0, 1))
                        opt.MouseEnter:Connect(function() TweenService:Create(opt, TweenInfo.new(0.15), {BackgroundColor3 = optHover}):Play() end)
                        opt.MouseLeave:Connect(function() TweenService:Create(opt, TweenInfo.new(0.15), {BackgroundColor3 = Color3.fromRGB(55,55,55)}):Play() end)
                    end

                    opt.MouseButton1Click:Connect(function()
                        selectedIndex = i
                        persistentState.Controls[generatedId] = selectedIndex
                        queueSave()
                        currentBtn.Text = tostring(v or "")
                        if type(callback) == "function" then pcall(function() callback(selectedIndex, v) end) end
                        toggle(false)
                    end)
                    table.insert(optionButtons, opt)
                end
            end
            rebuildOptions()
        
            local function computeHeaderAbsolute()
                local ok1, contPos = pcall(function() return container.AbsolutePosition end)
                local ok2, contSize = pcall(function() return container.AbsoluteSize end)
                if not ok1 or not ok2 then return Vector2.new(0,0), Vector2.new(0,0) end
                return contPos, contSize
            end
        
            local open = false
            local maxOpenHeight = 200
            local animTime = 0.16
        
            local function getTargetHeight()
                local contentH = listLayout.AbsoluteContentSize.Y + optionsPadding.PaddingTop.Offset + optionsPadding.PaddingBottom.Offset
                return math.clamp(contentH, 0, maxOpenHeight)
            end
        
            local currentTween = nil
            function toggle(state)
                if state == nil then state = not open end
                if state == open then return end
                open = state
                if open then
                    rebuildOptions()
                    local targetH = getTargetHeight()
                    local contPos, contSize = computeHeaderAbsolute()
                    local width = math.max(160, contSize.X - 16)
                    optionsFrame.Size = UDim2.new(0, width, 0, 0)
                    optionsFrame.Position = UDim2.new(0, contPos.X + 8, 0, contPos.Y + contSize.Y + 6)
                    optionsFrame.Visible = true
        
                    local tw = TweenService:Create(optionsFrame, TweenInfo.new(animTime, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.new(0, width, 0, getTargetHeight())})
                    currentTween = tw
                    tw:Play()
                else
                    local w = optionsFrame.Size.X.Offset
                    local tw = TweenService:Create(optionsFrame, TweenInfo.new(animTime, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.new(0, w, 0, 0)})
                    currentTween = tw
                    tw.Completed:Connect(function() optionsFrame.Visible = false end)
                    tw:Play()
                end
            end
        
            local updateConn
            local function startTracking()
                if updateConn then return end
                updateConn = RunService.RenderStepped:Connect(function()
                    if not open then return end
                    local contPos, contSize = computeHeaderAbsolute()
                    local width = math.max(160, contSize.X - 16)
                    optionsFrame.Position = UDim2.new(0, contPos.X + 8, 0, contPos.Y + contSize.Y + 6)
                    optionsFrame.Size = UDim2.new(0, width, 0, math.clamp(listLayout.AbsoluteContentSize.Y + optionsPadding.PaddingTop.Offset + optionsPadding.PaddingBottom.Offset, 0, maxOpenHeight))
                end)
            end
            local function stopTracking()
                if updateConn then updateConn:Disconnect(); updateConn = nil end
            end
        
            currentBtn.MouseButton1Click:Connect(function()
                toggle()
                if open then startTracking() else stopTracking() end
            end)
        
            local outsideConn
            outsideConn = UserInputService.InputBegan:Connect(function(inp)
                if not open then return end
                if inp.UserInputType == Enum.UserInputType.MouseButton1 or inp.UserInputType == Enum.UserInputType.Touch then
                    local mouse = UserInputService:GetMouseLocation()
                    local pos = optionsFrame.AbsolutePosition
                    local size = optionsFrame.AbsoluteSize
                    if mouse.X < pos.X or mouse.X > pos.X + size.X or mouse.Y < pos.Y or mouse.Y > pos.Y + size.Y then
                        toggle(false)
                        stopTracking()
                    end
                end
            end)
        
            local api = {}
            function api.Get() return selectedIndex, options[selectedIndex] end
            function api.SetIndex(i)
                i = math.clamp(i or 1, 1, #options)
                selectedIndex = i
                currentBtn.Text = tostring(options[selectedIndex] or "")
                rebuildOptions()
                if type(callback) == "function" then pcall(function() callback(selectedIndex, options[selectedIndex]) end) end
            end
            function api.SetValue(val)
                for i,v in ipairs(options) do if v == val then api.SetIndex(i); return end end
            end
            function api.UpdateOptions(newOptions, newDefaultIndex)
                options = newOptions or {}
                newDefaultIndex = math.clamp((newDefaultIndex or 1), 1, math.max(1,#options))
                selectedIndex = newDefaultIndex
                currentBtn.Text = options[selectedIndex] or "—"
                rebuildOptions()
            end
            api._frame = container
        
            local cleanup = function()
                if updateConn then updateConn:Disconnect(); updateConn = nil end
                if outsideConn then outsideConn:Disconnect(); outsideConn = nil end
                if optionsFrame and optionsFrame.Parent then optionsFrame:Destroy() end
            end
            container.AncestryChanged:Connect(function(_, parent) if not parent then cleanup() end end)
        
            return api
        end
        table.insert(self._tabs, tab)
        return tab
    end

    function window:Destroy()
        if self._gui and self._gui.Parent then self._gui:Destroy() end
    end

    -- [Improvement 1] Implementation of the Entrance Tween
    main.Parent = sg
    main.BackgroundTransparency = 1
    uiScale.Scale = 0.85
    TweenService:Create(main, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {BackgroundTransparency = 0}):Play()
    TweenService:Create(uiScale, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Scale = 1}):Play()

    return window
